set(wxWidgets_CONFIGURATION msw) # for MSW only

if (UNICODE)
   set(wxWidgets_CONFIGURATION "${wxWidgets_CONFIGURATION}u") # for MSW only
   set(wxWidgets_USE_UNICODE ON)
else ()
   set(wxWidgets_USE_UNICODE OFF)
endif ()

if (CMAKE_BUILD_TYPE STREQUAL "Debug")
   set(wxWidgets_USE_DEBUG ON)
   set(wxWidgets_CONFIGURATION "${wxWidgets_CONFIGURATION}d") # for MSW only
elseif (CMAKE_BUILD_TYPE STREQUAL "Release")
   set(wxWidgets_USE_DEBUG OFF)
endif ()

# pull in wxWidgets
find_package(wxWidgets)
if (wxWidgets_FOUND)
   include(${wxWidgets_USE_FILE})
else ()
   message("GrandOrgue requires wxWidgets!")
endif ()
